cmake_minimum_required(VERSION 3.25)

SET(PROJECTNAME celebrimbor)

PROJECT(${PROJECTNAME})

# Add .lib files
LINK_DIRECTORIES(${CMAKE_SOURCE_DIR}/lib)

FILE(GLOB_RECURSE PROJECT_SOURCES ../lib/*.cpp)

FILE(GLOB_RECURSE PROJECT_HEADERS ../include/*.h)

ADD_EXECUTABLE(${PROJECTNAME} ${PROJECT_HEADERS} ${PROJECT_SOURCES})

# OpenGL
FIND_PACKAGE(OpenGL REQUIRED)

# GLFW
find_package(GLFW3 REQUIRED)
message(STATUS "Found GLFW3 in ${GLFW3_INCLUDE_DIR}")

SET(LIBRARIES ${LIBRARIES} ${GLFW3_LIBRARY} ${OPENGL_LIBRARY})

INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/../include)

TARGET_LINK_LIBRARIES(${PROJECTNAME} ${LIBRARIES})

TARGET_COMPILE_OPTIONS(${PROJECTNAME} PUBLIC -std=c++14 -Wall -g)
